<range> Directive

For the report / inquire / process form which required range criteria selction. Click the button to to view the pageview for selecting a record.

Usage
- Use inside of entry
<range ng-model="deptRange">
	<label for="card_cardID">Department Range</label>
	<div class="form-inline">
	<div class="form-group">
	  <editbox ng-model="deptEditBoxStart" program-id="dw01dp" range="start" range-value="{{editboxCtrl.ngModel.DepartmentCode}}">
		<div class="input-group">
		  <input class="form-control" type="text" id="card_cardID" ng-model="rangeCtrl.ngModel.start" readonly>
		  <span class="input-group-btn"><button type="button" class="btn btn-default" ng-click="OpenPageView()"><i class="fa fa-bars"></i></button></span>
		</div>
	  </editbox>
	</div>
	<div class="form-group">
	  <editbox ng-model="deptEditBoxEnd" program-id="dw01dp" range="end" range-value="{{editboxCtrl.ngModel.DepartmentCode}}">
		<div class="input-group">
		  <input class="form-control" type="text" id="card_cardID" ng-model="rangeCtrl.ngModel.end" readonly>
		  <span class="input-group-btn"><button type="button" class="btn btn-default" ng-click="OpenPageView()"><i class="fa fa-bars"></i></button></span>
		</div>
	  </editbox>
	</div>
  </div>
</range>
ng-model = object
range = ["start" | "end"]
range-value = {{ expression }}
